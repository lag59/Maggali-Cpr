    // Business/topic dropdown logic
    const businessSelect = document.getElementById("businessSelect");
    const topicSelect = document.getElementById("topicSelect");
    const formStatus = document.getElementById("formStatus");

    const TOPICS = {
      cpr: [
        { value: "book-cpr", label: "Book CPR / BLS" },
        { value: "group-training", label: "Group Training (Construction Company)" },
        { value: "healthcare-facility", label: "Healthcare Facility Training" },
        { value: "question", label: "General Question" },
      ],
      design: [
        { value: "quote-design", label: "Request a Quote" },
        { value: "consult-design", label: "Book a Consult" },
        { value: "web-design", label: "Website Design" },
        { value: "dtf", label: "DTF Prints" },
        { value: "embroidery", label: "Embroidery" },
        { value: "business-cards", label: "Business Cards" },
        { value: "custom-print", label: "Custom Print Job" },
      ]
    };

    function loadTopics(mode, preferredValue) {
      topicSelect.innerHTML = "";
      TOPICS[mode].forEach(t => {
        const opt = document.createElement("option");
        opt.value = t.value;
        opt.textContent = t.label;
        topicSelect.appendChild(opt);
      });

      if (preferredValue) {
        const exists = [...topicSelect.options].some(o => o.value === preferredValue);
        if (exists) topicSelect.value = preferredValue;
      }
    }

    businessSelect?.addEventListener("change", () => {
      loadTopics(businessSelect.value);
      document.body.classList.toggle("modeDesign", businessSelect.value === "design");
    });

    document.querySelectorAll("[data-switch]").forEach(btn => {
      btn.addEventListener("click", () => {
        const mode = btn.getAttribute("data-switch");
        businessSelect.value = mode;
        loadTopics(mode);
        document.getElementById(mode)?.scrollIntoView({ behavior: "smooth" });
      });
    });

    document.querySelectorAll("[data-action]").forEach(el => {
      el.addEventListener("click", () => {
        const action = el.getAttribute("data-action");

        const isDesign = [
          "quote-design", "web-design", "dtf", "embroidery", "business-cards", "custom-print", "consult-design"
        ].includes(action);

        businessSelect.value = isDesign ? "design" : "cpr";
        loadTopics(businessSelect.value, action);

        if (formStatus) {
          const pretty = action.replaceAll("-", " ");
          formStatus.textContent = `Selected: ${businessSelect.value.toUpperCase()} → ${pretty}`;
          setTimeout(() => (formStatus.textContent = ""), 2500);
        }
      });
    });
